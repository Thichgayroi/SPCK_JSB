def bubble_sort(arr):
    for i in range(len(arr)):
        for j in range(len(arr) - 1 - i):
            if(arr[j] > arr[j+1]):
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# Ví dụ mảng
array = [5, 3, 8, 1, 2]

bubble_sort(array)

print(bubble_sort(array))
Time Complexity
O(n^2)
