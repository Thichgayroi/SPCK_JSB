def selection_sort(arr):
    for i in range(len(arr)-1):
        min_index = i
        for j in range(i + 1, len(arr)):
            if arr[j] < arr[min_index]:
                min_index = j
        arr[i], arr[min_index] = arr[min_index], arr[i]

# Ví dụ mảng
a = [5, 3, 8, 1, 2]

selection_sort(a)

print("Mảng sau khi sắp xếp:", a)
