def merge_sort(arr):
    if len(arr) > 1:
        mid = len(arr) // 2  # Chia đôi mảng
        left = arr[:mid]
        right = arr[mid:]

        # Đệ quy sắp xếp 2 nửa
        merge_sort(left)
        merge_sort(right)

        # Trộn hai nửa lại
        i = j = k = 0
        while i < len(left) and j < len(right):
            if left[i] < right[j]:
                arr[k] = left[i]
                i += 1
            else:
                arr[k] = right[j]
                j += 1
            k += 1

        # Thêm các phần tử còn lại
        while i < len(left):
            arr[k] = left[i]
            i += 1
            k += 1

        while j < len(right):
            arr[k] = right[j]
            j += 1
            k += 1

# Ví dụ mảng
arr = [38, 27, 43, 3, 9, 82, 10]
merge_sort(arr)
print("Mảng sau khi sắp xếp:", arr)
